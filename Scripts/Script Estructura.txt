CREACIÓN DE LAS TABLAS 

-- Tabla Membresias
CREATE TABLE Membresias (
id_membresia NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
nombre VARCHAR2(50) NOT NULL,
precio NUMBER(10, 2) NOT NULL,
duracion NUMBER NOT NULL
);

-- Tabla Clientes
CREATE TABLE Clientes (
id_cliente NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
nombre VARCHAR2(50) NOT NULL,
apellido_paterno VARCHAR2(50) NOT NULL,
apellido_materno VARCHAR2(50) NOT NULL,
edad NUMBER NOT NULL,
email VARCHAR2(100) UNIQUE NOT NULL,
telefono VARCHAR2(15) UNIQUE NOT NULL,
-- FKs
id_membresia NUMBER,
CONSTRAINT fk_cliente_membresia
FOREIGN KEY (id_membresia) 
REFERENCES Membresias(id_membresia)
);

-- Tabla Evaluaciones
CREATE TABLE Evaluaciones (
id_evaluacion NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
peso NUMBER(5, 2) NOT NULL,
grasa_corporal NUMBER(5, 2) NOT NULL,
masa_muscular NUMBER(5, 2) NOT NULL,
fecha_evaluacion DATE NOT NULL,
-- FKs
id_cliente NUMBER NOT NULL,
CONSTRAINT fk_evaluacion_cliente
FOREIGN KEY (id_cliente)
REFERENCES Clientes(id_cliente)
);

-- Tabla Pagos
CREATE TABLE Pagos (
id_pago NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
monto NUMBER(10, 2) NOT NULL,
fecha_pago DATE NOT NULL,
metodo_pago VARCHAR2(10) CHECK (metodo_pago IN ('Tarjeta', 'Efectivo')) NOT NULL,
concepto VARCHAR2(255),
-- FKs
id_cliente NUMBER NOT NULL,
CONSTRAINT fk_pago_cliente
FOREIGN KEY (id_cliente)
REFERENCES Clientes(id_cliente)
);

-- Tabla Entrenadores
CREATE TABLE Entrenadores (
id_entrenador NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
nombre VARCHAR2(50) NOT NULL,
apellido_paterno VARCHAR2(50) NOT NULL,
apellido_materno VARCHAR2(50) NOT NULL,
especialidad VARCHAR2(100)
);

-- Tabla Clases
CREATE TABLE Clases (
id_clase NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
nombre VARCHAR2(50) NOT NULL,
descripcion VARCHAR2(255),
capacidad NUMBER,
horario VARCHAR2(50),
-- FKs
id_entrenador NUMBER,
CONSTRAINT fk_clase_entrenador
FOREIGN KEY (id_entrenador)
REFERENCES Entrenadores(id_entrenador)
);

-- Tabla Inscripciones
CREATE TABLE Inscripciones (
id_inscripcion NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
fecha_inscripcion DATE NOT NULL,
-- FKs
id_cliente NUMBER NOT NULL,
id_clase NUMBER NOT NULL,
CONSTRAINT fk_inscripcion_cliente
FOREIGN KEY (id_cliente)
REFERENCES Clientes(id_cliente),
CONSTRAINT fk_inscripcion_clase
FOREIGN KEY (id_clase) 
REFERENCES Clases(id_clase)
);

-- Tabla Proveedores
CREATE TABLE Proveedores (
id_proveedor NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
nombre VARCHAR2(50) NOT NULL,
telefono VARCHAR2(15) UNIQUE NOT NULL,
direccion VARCHAR2(255) UNIQUE NOT NULL
);

-- Tabla Productos
CREATE TABLE Productos (
id_producto NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
nombre VARCHAR2(50) NOT NULL,
precio NUMBER(10, 2) NOT NULL,
descripcion VARCHAR2(255),
existencias NUMBER NOT NULL,
--  FKs
id_proveedor NUMBER NOT NULL,
CONSTRAINT fk_producto_proveedor 
FOREIGN KEY (id_proveedor) 
REFERENCES Proveedores(id_proveedor)
);

-- Tabla Pedidos
CREATE TABLE Pedidos (
id_pedido NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
fecha_entrega DATE,
estado VARCHAR2(15) CHECK (estado IN ('Entregado', 'No entregado', 'En espera', 'En camino')),
-- FKs
id_cliente NUMBER NOT NULL,
CONSTRAINT fk_pedido_cliente 
FOREIGN KEY (id_cliente) 
REFERENCES Clientes(id_cliente)
);

-- Tabla PedidoProductos
CREATE TABLE PedidoProductos (
id_pedido_productos NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
cantidad NUMBER NOT NULL,
-- FKs
id_pedido NUMBER NOT NULL,
id_producto NUMBER NOT NULL,
CONSTRAINT fk_pedidoproducto_pedido 
FOREIGN KEY (id_pedido) 
REFERENCES Pedidos(id_pedido),
CONSTRAINT fk_pedidoproducto_producto 
FOREIGN KEY (id_producto) 
REFERENCES Productos(id_producto)
);
